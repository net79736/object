package test.claude6_2;

import java.sql.Date;
import java.util.List;


public class ReportGenerator {
    public String generateSalesReport(Date startDate, Date endDate) {
        // OrderRepository를 사용하여 주문 목록을 조회합니다.
        OrderRepository orderRepository = new OrderRepository();
        // 주문 목록을 조회합니다.
        List<Order> orders = orderRepository.findByDateRange(startDate, endDate);
        
        // 주문 목록을 순회하며 총 매출을 계산합니다.
        double totalSales = 0;
        
        for (Order order : orders) {
            totalSales += order.calculateTotalAmount(orders);
        }

        int totalOrders = orders.size();
        double averageOrderValue = Order.calculateAverageOrderValue(totalSales, totalOrders);
        
        return printReport(totalSales, totalOrders, averageOrderValue);
    }
    
    // 보고서를 생성합니다.
    public String printReport(double totalSales, int totalOrders, double averageOrderValue) {
        return String.format("총 매출: %.2f원\n주문 건수: %d건\n평균 주문액: %.2f원", 
                           totalSales, totalOrders, averageOrderValue));
    }
}
